    ensureBaseLayer();

    const curMinY = 0;
    const curMaxX = base.canvas.width;

    for (let y = Math.max(0, cy - radius); y <= Math.min(refSnapH - 1, cy + radius); y++) {

function drawPath(ctx, pts, smooth = false) {
    if (!pts.length) return;
    ctx.beginPath();
    ctx.moveTo(pts[0].x, pts[0].y);
    if (!smooth || pts.length < 3) {
        for (let i = 1; i < pts.length; i++) ctx.lineTo(pts[i].x, pts[i].y);
        return;
    }
    for (let i = 1; i < pts.length - 1; i++) {
        const mx = (pts[i].x + pts[i + 1].x) / 2;
        const my = (pts[i].y + pts[i + 1].y) / 2;
        ctx.quadraticCurveTo(pts[i].x, pts[i].y, mx, my);
    }
    const last = pts[pts.length - 1];
    ctx.lineTo(last.x, last.y);
}

function strokeBounds(pts) {
    let minX = Infinity, minY = Infinity, maxX = -Infinity, maxY = -Infinity;
    for (const p of pts) {
        if (p.x < minX) minX = p.x;
        if (p.y < minY) minY = p.y;
        if (p.x > maxX) maxX = p.x;
        if (p.y > maxY) maxY = p.y;
    }
    return { minX, minY, maxX, maxY, w: Math.max(1, maxX - minX), h: Math.max(1, maxY - minY) };
}

function pathLength(pts) {
    let len = 0;
    for (let i = 1; i < pts.length; i++) len += Math.hypot(pts[i].x - pts[i - 1].x, pts[i].y - pts[i - 1].y);
    return len;
}

function resampleStroke(pts, n = 24) {
    if (!pts.length) return [];
    if (pts.length === 1) return Array.from({ length: n }, () => ({ x: pts[0].x, y: pts[0].y }));
    const total = pathLength(pts);
    if (total === 0) return Array.from({ length: n }, () => ({ x: pts[0].x, y: pts[0].y }));
    const step = total / (n - 1);
    const out = [{ x: pts[0].x, y: pts[0].y }];
    let dist = 0;
    let prev = pts[0];
    for (let i = 1; i < pts.length; i++) {
        let curr = pts[i];
        let seg = Math.hypot(curr.x - prev.x, curr.y - prev.y);
        if (seg === 0) continue;
        while ((dist + seg) >= step && out.length < n) {
            const t = (step - dist) / seg;
            const x = prev.x + t * (curr.x - prev.x);
            const y = prev.y + t * (curr.y - prev.y);
            out.push({ x, y });
            prev = { x, y };
            seg = Math.hypot(curr.x - prev.x, curr.y - prev.y);
            dist = 0;
        }
        dist += seg;
        prev = curr;
    }
    while (out.length < n) out.push({ x: pts[pts.length - 1].x, y: pts[pts.length - 1].y });
    return out;
}

function normalizeStroke(pts, n = 24) {
    const sample = resampleStroke(pts, n);
    const b = strokeBounds(sample);
    return {
        points: sample.map(p => ({
            x: (p.x - b.minX) / b.w,
            y: (p.y - b.minY) / b.h
        })),
        bounds: b,
        aspect: b.w / b.h,
    };
}

function denormalizeStroke(norm, bounds) {
    return norm.points.map(p => ({
        x: bounds.minX + p.x * bounds.w,
        y: bounds.minY + p.y * bounds.h
    }));
}

function strokeDistance(a, b) {
    const n = Math.min(a.length, b.length);
    let d = 0;
    for (let i = 0; i < n; i++) d += Math.hypot(a[i].x - b[i].x, a[i].y - b[i].y);
    return d / Math.max(1, n);
}

function maybeLearnStroke(pts) {
    if (!learnOn || pts.length < 5) return pts;
    const sig = normalizeStroke(pts, 24);
    let bestIndex = -1;
    let bestScore = Infinity;
    let bestReverse = false;
    for (let i = 0; i < strokeMemory.length; i++) {
        const mem = strokeMemory[i];
        const forward = strokeDistance(sig.points, mem.points);
        const reverse = strokeDistance(sig.points, mem.points.slice().reverse());
        const score = Math.min(forward, reverse);
        if (score < bestScore) {
            bestScore = score;
            bestIndex = i;
            bestReverse = reverse < forward;
        }
    }
    const out = pts.slice();
    if (bestIndex >= 0 && bestScore < 0.14) {
        const mem = strokeMemory[bestIndex];
        const fit = denormalizeStroke({
            points: bestReverse ? mem.points.slice().reverse() : mem.points
        }, strokeBounds(pts));
        return fit;
    }
    if (strokeMemory.length >= MAX_STROKE_MEM) strokeMemory.shift();
    strokeMemory.push(sig);
    $('svMem').textContent = strokeMemory.length;
    return out;
}

    const last = points[points.length - 1];
    if (Math.hypot(lp.x - last.x, lp.y - last.y) < 0.8 / vpZ) return;
    points.push(lp); renderTemp();
}, { passive: false });

window.addEventListener('touchend', e => {
    if (e.touches.length > 0) return;
    if (panning) { panning = false; delete cvTemp.dataset.pan; return; }
    if (!drawing) return;
    drawing = false; ctxTemp.clearRect(0, 0, W, H);
    if (points.length < 3) { points = []; return; }
    const base = mode === 'erase' ? points : correctShape(points);
    commitStroke(mode === 'erase' ? base : maybeLearnStroke(base));
    points = [];
});

// ── commit ──
function commitStroke(pts) {
    if (pts.length < 2) return;
    const layer = activeLayer();
    if (!layer) return;
    let drawPts = pts;
    const minX = Math.min(...pts.map(p => p.x));
    const minY = Math.min(...pts.map(p => p.y));
    const maxX = Math.max(...pts.map(p => p.x));
    const maxY = Math.max(...pts.map(p => p.y));
    const pad = Math.max(96, (mode === 'erase' ? eraserSize : brushSize) * 4);
    const growth = ensureWorldRect(minX - pad, minY - pad, maxX + pad, maxY + pad, pad);
    if (growth?.expanded) {
        drawPts = pts.map(p => ({ x: p.x + growth.dx, y: p.y + growth.dy }));
    }
    if (layer.history.length >= MAX_HIST) layer.history.shift();
    layer.history.push(layer.ctx.getImageData(0, 0, layer.canvas.width, layer.canvas.height));
    $('svHist').textContent = layer.history.length;
    $('tbUndo').disabled = false;
    applyStyle(layer.ctx);
    drawPath(layer.ctx, drawPts, mode === 'pencil');
    const f = drawPts[0], l = drawPts[drawPts.length - 1];
    if (pts.length > 4 && Math.hypot(f.x - l.x, f.y - l.y) < 18) layer.ctx.closePath();
    layer.ctx.stroke();
    layer.ctx.globalCompositeOperation = 'source-over'; layer.ctx.globalAlpha = 1;
    renderDraw();
}

// ── corrección ──
function correctShape(pts) {
    if (!assistOn || pts.length < 6) return pts;
    const s = pts[0], e = pts[pts.length - 1];
    const len = Math.hypot(e.x - s.x, e.y - s.y);
    if (len > 15 && pts.length > 5) {
        let md = 0;
        for (const p of pts) { const d = Math.abs((e.y - s.y)*p.x - (e.x - s.x)*p.y + e.x*s.y - e.y*s.x) / len; md = Math.max(md, d); }
        if (md < Math.max(5, len * 0.07)) { bump('📏 Línea perfecta'); return [s, e]; }
    }
    if (pts.length > 15) {
        const cx = pts.reduce((a,p)=>a+p.x,0)/pts.length, cy = pts.reduce((a,p)=>a+p.y,0)/pts.length;
        const ds = pts.map(p=>Math.hypot(p.x-cx,p.y-cy));
        const r = ds.reduce((a,v)=>a+v,0)/ds.length, vr = ds.reduce((a,v)=>a+Math.abs(v-r),0)/ds.length;
        if (vr < r*0.22 && r > 12) { bump('⚪ Círculo'); return Array.from({length:61},(_,i)=>({x:cx+r*Math.cos(i/60*Math.PI*2),y:cy+r*Math.sin(i/60*Math.PI*2)})); }
    }
    return pts;
}
function bump(msg) { corrections++; $('svCorr').textContent = corrections; }

// ── Loomis ──
function drawLoomis(ctx) {
    const cx=W/2, top=H*.07, hw=W*.24, fh=H*.6;
    ctx.save(); ctx.setLineDash([4,4]); ctx.lineWidth=1/vpZ; ctx.globalAlpha=.45;
    ctx.strokeStyle='#60a5fa'; ctx.beginPath(); ctx.ellipse(cx,top+fh*.38,hw,fh*.38,0,0,Math.PI*2); ctx.stroke();
    ctx.beginPath(); ctx.moveTo(cx-hw*.75,top+fh*.62); ctx.quadraticCurveTo(cx,top+fh,cx+hw*.75,top+fh*.62); ctx.stroke();
    ctx.strokeStyle='#a78bfa'; ctx.beginPath(); ctx.moveTo(cx,top); ctx.lineTo(cx,top+fh); ctx.stroke();
    [[fh*.42,'#39d353',.95],[fh*.65,'#fdba74',.8],[fh*.8,'#f778ba',.65]].forEach(([y,c,w])=>{
        ctx.strokeStyle=c; ctx.beginPath(); ctx.moveTo(cx-hw*w,top+y); ctx.lineTo(cx+hw*w,top+y); ctx.stroke();
    });
    ctx.restore();
}

    const next = createLayer('Canvas ' + (layers.length + 1));

    activeLayerIndex += 1;
    refreshLayerUI();

function deleteLayer() {
    if (layers.length <= 1) return;
    layers.splice(activeLayerIndex, 1);
    activeLayerIndex = Math.max(0, activeLayerIndex - 1);
    refreshLayerUI();
    renderDraw();
}

function moveLayer(dir) {
    const i = activeLayerIndex;
    const j = i + dir;
    if (j < 0 || j >= layers.length) return;
    [layers[i], layers[j]] = [layers[j], layers[i]];
    activeLayerIndex = j;
    refreshLayerUI();
    renderDraw();
}

function applyEdgePreviewToMain() {
    if (edgeBusy) return;
    edgeBusy = true;
    try {
        const prev = $('edgePrev');
        if (!prev.classList.contains('show')) return;
        const layer = activeLayer();
        if (!layer) return;
        const tmp = document.createElement('canvas');
        tmp.width = prev.width;
        tmp.height = prev.height;
        const tctx = tmp.getContext('2d');
        tctx.drawImage(prev, 0, 0);
        const data = tctx.getImageData(0, 0, tmp.width, tmp.height);
        for (let i = 0; i < data.data.length; i += 4) {
            const r = data.data[i];
            const g = data.data[i + 1];
            const b = data.data[i + 2];
            const brightness = (r + g + b) / 3;
            if (brightness > 245) {
                data.data[i + 3] = 0;
            } else {
                data.data[i] = 0;
                data.data[i + 1] = 0;
                data.data[i + 2] = 0;
                data.data[i + 3] = 255;
            }
        }
        tctx.putImageData(data, 0, 0);
        if (layer.history.length >= MAX_HIST) layer.history.shift();
        layer.history.push(layer.ctx.getImageData(0, 0, layer.canvas.width, layer.canvas.height));
        $('svHist').textContent = layer.history.length;
        $('tbUndo').disabled = false;
        ensureWorldRect(imgX, imgY, imgX + imgW, imgY + imgH, 64);
        layer.ctx.save();
        layer.ctx.globalCompositeOperation = 'source-over';
        layer.ctx.drawImage(tmp, imgX, imgY, imgW, imgH);
        layer.ctx.restore();
        renderDraw();
    } finally {
        edgeBusy = false;
    }
}

    try{
        const tmp=document.createElement('canvas');
        const maxSide = Math.max(imgW, imgH) || 1;
        const scale = Math.min(1, EDGE_MAX_SIDE / maxSide);
        tmp.width=Math.max(1, Math.round(imgW * scale));
        tmp.height=Math.max(1, Math.round(imgH * scale));
        tmp.getContext('2d').drawImage(img,0,0,tmp.width,tmp.height);
        const src=cv.imread(tmp),gray=new cv.Mat(),blur=new cv.Mat(),dst=new cv.Mat();
        cv.cvtColor(src,gray,cv.COLOR_RGBA2GRAY);
        cv.GaussianBlur(gray,blur,new cv.Size(3,3),0);
        cv.Canny(blur,dst,50,150); cv.bitwise_not(dst,dst);
        const prev=$('edgePrev');
        prev.width=tmp.width; prev.height=tmp.height;
        cv.imshow(prev,dst); prev.classList.add('show');
        src.delete();gray.delete();blur.delete();dst.delete();
        $('btnApplyEdge').disabled=false;
    }catch(e){console.error(e);}
}
$('btnApplyEdge').addEventListener('click',()=>{
    const prev=$('edgePrev');
    if(!prev.classList.contains('show')) return;
    if(!window.confirm('Aplicar autocalco sobre el dibujo?')) return;
    applyEdgePreviewToMain();
});

let faceMesh=null;
function initMediaPipe(){
    if(typeof FaceMesh==='undefined'){ setTimeout(initMediaPipe,1500); return; }
    try{
        faceMesh=new FaceMesh({locateFile:f=>`https://cdn.jsdelivr.net/npm/@mediapipe/face_mesh/${f}`});
        faceMesh.setOptions({maxNumFaces:1,refineLandmarks:true,minDetectionConfidence:.5,minTrackingConfidence:.5});
        faceMesh.onResults(r=>{
            if(!r.multiFaceLandmarks?.length) return;
            const lms=r.multiFaceLandmarks[0];
            const lx=i=>imgX+lms[i].x*imgW, ly=i=>imgY+lms[i].y*imgH;
            const layer = activeLayer();
            if (!layer) return;
            ensureWorldRect(imgX, imgY, imgX + imgW, imgY + imgH, 64);
            layer.ctx.save(); layer.ctx.setLineDash([3,3]); layer.ctx.globalAlpha=.5; layer.ctx.lineWidth=1;
            const ol=[10,338,297,332,284,251,389,356,454,323,361,288,397,365,379,378,400,377,152,148,176,149,150,136,172,58,132,93,234,127,162,21,54,103,67,109,10];
            layer.ctx.strokeStyle='#60a5fa'; layer.ctx.beginPath(); layer.ctx.moveTo(lx(ol[0]),ly(ol[0])); ol.slice(1).forEach(i=>layer.ctx.lineTo(lx(i),ly(i))); layer.ctx.closePath(); layer.ctx.stroke();
            layer.ctx.strokeStyle='#39d353'; layer.ctx.beginPath(); layer.ctx.moveTo(lx(33),ly(33)); layer.ctx.lineTo(lx(263),ly(263)); layer.ctx.stroke();
            layer.ctx.strokeStyle='#f778ba'; layer.ctx.beginPath(); layer.ctx.moveTo(lx(61),ly(61)); layer.ctx.lineTo(lx(291),ly(291)); layer.ctx.stroke();
            layer.ctx.restore(); renderDraw();
        });
        plugins.mp=true;
        $('pillMP').className='pill ready';
        $('badgeMP').className='bdg ready';
    }catch(e){ $('pillMP').className='pill err'; }
}
